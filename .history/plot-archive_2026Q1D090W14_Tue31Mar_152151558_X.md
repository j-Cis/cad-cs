```rust
# Plik konfiguracyjny CD (Continuous Deployment) dla narzędzia cad-cs.
# Automatyzuje proces wydań po dodaniu taga wersji (np. v0.1.2).
# Kompiluje i pakuje binarki dla Windows, Linux i macOS do sekcji GitHub Releases.
# Wywołanie: git tag v0.1.2 && git push origin v0.1.2

name: Deploy Release

on:
  push:
    tags:
      - 'v*'

permissions:
  contents: write
  id-token: write # DODANE: Kluczowe dla Trusted Publishing na crates.io

jobs:
  # --- NOWE ZADANIE SPRAWDZAJĄCE GAŁĄŹ ---
  check-branch:
    name: Check if tag is on main
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0 # Wymagane, aby pobrać wszystkie gałęzie i historię
      
      - name: Verify Branch
        run: |
          # Pobiera SHA commita, do którego przypisany jest tag
          COMMIT_SHA=${{ github.sha }}
          
          # Sprawdza, czy ten commit należy do gałęzi origin/main
          if git branch -r --contains "$COMMIT_SHA" | grep -q "origin/main"; then
            echo "Tag is on main. Proceeding with release."
            exit 0
          else
            echo "Error: Tag is NOT on main. It belongs to another branch (e.g., next)."
            exit 1
          fi

  build-and-upload:
    name: Build ${{ matrix.target }}
    needs: check-branch # <-- Kluczowe: czeka na sukces zadania sprawdzającego
    runs-on: ${{ matrix.os }}
    strategy:
      fail-fast: false
      matrix:
        include:
          # --- WINDOWS ---
          - target: x86_64-pc-windows-msvc
            os: windows-latest
            archive_name: cad-cs-x86_64-windows.zip
            bin_name: cad-cs.exe
          
          - target: i686-pc-windows-msvc
            os: windows-latest
            archive_name: cad-cs-i686-windows.zip
            bin_name: cad-cs.exe

          # --- LINUX ---
          - target: x86_64-unknown-linux-musl
            os: ubuntu-latest
            archive_name: cad-cs-x86_64-linux.tar.gz
            bin_name: cad-cs
            use_musl: true

          - target: i686-unknown-linux-gnu
            os: ubuntu-latest
            archive_name: cad-cs-i686-linux.tar.gz
            bin_name: cad-cs
            install_multilib: true

          # --- MACOS ---
          - target: aarch64-apple-darwin
            os: macos-latest
            archive_name: cad-cs-aarch64-macos.tar.gz
            bin_name: cad-cs

          - target: x86_64-apple-darwin
            os: macos-latest
            archive_name: cad-cs-x86_64-macos.tar.gz
            bin_name: cad-cs

    steps:
      - uses: actions/checkout@v4

      - name: Install Rust
        run: rustup update stable && rustup target add ${{ matrix.target }}

      # --- ZALEŻNOŚCI LINUX ---
      - name: Install Linux Deps
        if: runner.os == 'Linux'
        run: sudo apt-get update && sudo apt-get install -y build-essential
          
      - name: Install Multilib (Linux 32-bit)
        if: matrix.install_multilib == true
        run: sudo apt-get install -y gcc-multilib

      - name: Install MUSL Tools
        if: matrix.use_musl == true
        run: sudo apt-get install -y musl-tools

      # --- BUDOWANIE ---
      - name: Build
        run: cargo build --release --target ${{ matrix.target }} --verbose

      # --- PAKOWANIE (WINDOWS -> ZIP) ---
      - name: Package Windows (Zip)
        if: runner.os == 'Windows'
        run: |
          New-Item -ItemType Directory -Force -Path staging
          Copy-Item -Path "target/${{ matrix.target }}/release/${{ matrix.bin_name }}" -Destination "staging/${{ matrix.bin_name }}"
          Copy-Item -Path "README.md" -Destination "staging/README.md"
          Compress-Archive -Path staging/* -DestinationPath ${{ matrix.archive_name }}

      # --- PAKOWANIE (LINUX/MAC -> TAR.GZ) ---
      - name: Package Unix (Tar)
        if: runner.os != 'Windows'
        run: |
          mkdir staging
          cp target/${{ matrix.target }}/release/${{ matrix.bin_name }} staging/${{ matrix.bin_name }}
          cp README.md staging/README.md
          chmod +x staging/${{ matrix.bin_name }}
          tar -czvf ${{ matrix.archive_name }} -C staging .

      # --- PUBLIKACJA ---
      - name: Release
        uses: softprops/action-gh-release@v1
        with:
          files: ${{ matrix.archive_name }}
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

  # DODANE: Nowe zadanie wysyłające paczkę na crates.io
  publish-to-crates-io:
    name: Publish to crates.io
    runs-on: ubuntu-latest
    needs: [check-branch, build-and-upload] # Czeka na zakończenie obu poprzednich
    steps:
      - uses: actions/checkout@v4
      - name: Install Rust
        run: rustup update stable
      - name: Publish
        env:
          CARGO_REGISTRY_TOKEN: ${{ secrets.CARGO_REGISTRY_TOKEN }}
        run: cargo publish

```

### 002: `./.github/workflows/rust.yml`

```rust
# Plik konfiguracyjny CI (Continuous Integration) dla narzędzia cad-cs.
# Odpowiada za automatyczne sprawdzanie formatowania, lintowanie (Clippy) oraz testy.
# Buduje wersje binarne dla systemów Windows i Linux przy każdym wypchnięciu kodu.
# Zapewnia wysoką jakość i stabilność kodu źródłowego w gałęzi głównej.

name: Rust CI & Build

on:
  push:
    branches: [ "main" ]
  pull_request:
    branches: [ "main" ]
  workflow_dispatch:

env:
  CARGO_TERM_COLOR: always

jobs:
  build-and-test:
    name: Build on ${{ matrix.os }}
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [ubuntu-latest, windows-latest]

    steps:
    - uses: actions/checkout@v4

    # Gwarantuje obecność clippy i rustfmt
    - name: Update Rust Toolchain
      run: rustup update stable && rustup default stable && rustup component add clippy rustfmt

    # Zależności graficzne niezbędne do kompilacji Slint GUI na Linuxie
    - name: Install Linux Dependencies
      if: runner.os == 'Linux'
      run: |
        sudo apt-get update
        sudo apt-get install -y libxcb-render0-dev libxcb-shape0-dev libxcb-xfixes0-dev libx11-dev libssl-dev libfontconfig1-dev libxkbcommon-dev

    # Pamięć podręczna w najnowszej wersji v4 (przyspiesza budowanie)
    - name: Cache Cargo
      uses: actions/cache@v4
      with:
        path: |
          ~/.cargo/bin/
          ~/.cargo/registry/index/
          ~/.cargo/registry/cache/
          ~/.cargo/git/db/
          target/
        key: ${{ runner.os }}-cargo-${{ hashFiles('**/Cargo.lock') }}

    # 1. Sprawdzanie formatowania
    - name: Check Formatting
      run: cargo fmt -- --check

    # 2. Sprawdzanie jakości kodu (Linter Clippy)
    - name: Lint with Clippy
      run: cargo clippy -- -D warnings

    # 3. Uruchomienie testów
    - name: Run tests
      run: cargo test --verbose

    # 4. Budowanie wersji produkcyjnej (Release)
    - name: Build Release
      run: cargo build --release --verbose

    # 5. Wyciągnięcie gotowego pliku .exe (Windows)
    - name: Upload Windows Artifact
      if: runner.os == 'Windows'
      uses: actions/upload-artifact@v4
      with:
        name: cad-cs-windows
        path: target/release/cad-cs.exe

    # 6. Wyciągnięcie gotowego pliku binarnego (Linux)
    - name: Upload Linux Artifact
      if: runner.os == 'Linux'
      uses: actions/upload-artifact@v4
      with:
        name: cad-cs-linux
        path: target/release/cad-cs

```
